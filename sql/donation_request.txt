CREATE TABLE `digitalbloodbank`.`donation_request` ( `SrNo` INT(10) NOT NULL AUTO_INCREMENT , `name` VARCHAR(128) NOT NULL , `gender` VARCHAR(10) NOT NULL , `DOB` DATE NOT NULL , `contact_no` INT(12) NOT NULL , `address` VARCHAR(256) NOT NULL , `tdate` DATE NOT NULL , `state` VARCHAR(128) NOT NULL , `district` VARCHAR(128) NOT NULL , `bid` VARCHAR(128) NOT NULL , `blood_group` VARCHAR(10) NOT NULL , PRIMARY KEY (`SrNo`)) ENGINE = InnoDB;
